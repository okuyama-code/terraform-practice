```ruby
# AWSのセキュリティグループリソースを定義
resource "aws_security_group" "example_ec2" {
  # セキュリティグループの名前
  name = "example_ec2"

  # インバウンドルール（外部からのアクセスを許可するルール）
  ingress {
    # 許可するポート番号の範囲（開始）
    from_port = 80
    # 許可するポート番号の範囲（終了）
    to_port = 80
    # 許可するプロトコル（ここではTCP）
    protocol = "tcp"
    # アクセスを許可するIPアドレスの範囲（ここでは全てのIPアドレス）
    cidr_blocks = ["0.0.0.0/0"]
  }

  # アウトバウンドルール（外部へのアクセスを許可するルール）
  egress {
    # 許可するポート番号の範囲（開始）
    from_port = 0
    # 許可するポート番号の範囲（終了）
    to_port = 0
    # 許可するプロトコル（ここでは全てのプロトコル）
    protocol = "-1"
    # アクセスを許可するIPアドレスの範囲（ここでは全てのIPアドレス）
    cidr_blocks = ["0.0.0.0/0"]
  }
}
```
